2.14 Interval problems

Mental model. Three transformations cover most interval problems: 1. Sort by start — for merging. 2. Sort by end — for greedy scheduling (2.8). 3. Sweep line / events — sort (point, +1/−1), scan, accumulate.

Example (merge intervals).

def merge(iv):
    iv.sort()
    out = []
    for s, e in iv:
        if out and s <= out[-1][1]: out[-1][1] = max(out[-1][1], e)
        else: out.append([s, e])
    return out

function merge(iv) {
  iv.sort((a, b) => a[0] - b[0] || a[1] - b[1]);
  const out = [];
  for (const [s, e] of iv) {
    if (out.length && s <= out[out.length - 1][1]) {
      out[out.length - 1][1] = Math.max(out[out.length - 1][1], e);
    } else out.push([s, e]);
  }
  return out;
}

Example (meeting rooms II — sweep line).

def min_rooms(iv):
    ev = []
    for s, e in iv: ev.append((s, +1)); ev.append((e, -1))
    ev.sort()
    cur, best = 0, 0
    for _, d in ev: cur += d; best = max(best, cur)
    return best

function minRooms(iv) {
  const ev = [];
  for (const [s, e] of iv) { ev.push([s, +1]); ev.push([e, -1]); }
  ev.sort((a, b) => a[0] - b[0] || a[1] - b[1]);
  let cur = 0, best = 0;
  for (const [, d] of ev) { cur += d; best = Math.max(best, cur); }
  return best;
}
